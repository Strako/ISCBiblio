CREATE TRIGGER decrement_quantity
AFTER INSERT ON borrows FOR EACH ROW
UPDATE books SET books.quantity = books.quantity - 1 WHERE book_id = new.book_id;


CREATE TRIGGER increment_quantity
AFTER UPDATE ON borrows FOR EACH ROW
UPDATE books SET books.quantity = books.quantity + 1 WHERE book_id = new.book_id;
